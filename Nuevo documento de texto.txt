  <main class="card">
    <h1>Patrón Factory Method (versión sencilla en frontend)</h1>
    <p class="small">
      <strong>Idea:</strong> Un <em>Factory</em> decide qué controlador crear según el tipo seleccionado.
      Así se organiza mejor la lógica y es fácil agregar nuevos tipos sin tocar el resto del código.
    </p>

    <div class="grid">
      <button class="btn accent" data-type="users">Controlador de usuarios</button>
      <button class="btn accent" data-type="products">Controlador de productos</button>
      <button class="btn" data-type="orders">Controlador de pedidos (nuevo)</button>
      <button class="btn" data-type="unknown">Tipo desconocido</button>
    </div>

    <div class="output" id="output">Salida del controlador aquí…</div>

    <p class="small">
      <strong>Patrón aplicado:</strong> <code>ControllerFactory.getController(type)</code> devuelve una instancia acorde al
      <em>type</em>. Cada controlador implementa <code>handle()</code> y puede tener su propia lógica.
    </p>
  </main>

  <script>
    // Controladores simples con la misma interfaz (handle)
    class UserController {
      handle() {
        return "Usuarios: listar, crear, actualizar, eliminar.";
      }
    }

    class ProductController {
      handle() {
        return "Productos: catálogo, precios, stock.";
      }
    }

    class OrderController {
      handle() {
        return "Pedidos: carrito, checkout, historial.";
      }
    }

    // Factory Method: decide qué instancia crear según 'type'
    class ControllerFactory {
      static getController(type) {
        switch (type) {
          case "users": return new UserController();
          case "products": return new ProductController();
          case "orders": return new OrderController();
          default:
            throw new Error("Tipo de controlador no válido: " + type);
        }
      }
    }

    // UI: manejar clics y mostrar la salida del controlador
    const output = document.getElementById("output");
    document.querySelectorAll("button[data-type]").forEach(btn => {
      btn.addEventListener("click", () => {
        const type = btn.getAttribute("data-type");
        try {
          const controller = ControllerFactory.getController(type);
          const result = controller.handle();
          output.textContent = `Factory devolvió '${type}': ${result}`;
        } catch (err) {
          output.textContent = "Error: " + err.message;
        }
      });
    });
  </script>
